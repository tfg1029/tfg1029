📜 制作ルール
  • サイトの方針
    • 商品はほぼ一点もののため、Webサイトでの商品販売は行わない
    • 実店舗への来店促進がWebサイトの目的
  • ファイル管理
    • 正式ライン：0125, 0126…（四桁）で連番管理。現行公開サイトの継続改修に使用
    • ベータライン：b0001, b0002…（四桁）で管理。新デザイン・新構成の実験に使用。完成・承認されたら正式ラインに移行（目安2〜3ヶ月）
    • 変更のたびに新しい番号で保存
    • 管理資料ファイル名：鉱石のすみかWeb管理資料_v####.html（例：v0044→v0045）※四桁
  • 進め方
    • いきなりコーディングしない。提案 → OK確認 → コーディングの順で進める
    • 提案前に現状・仕様・前提条件を確認・整理する
    • 提案時はメリット・デメリット・影響範囲を明示する
    • 指示された箇所以外は勝手に変更しない。改善点があれば根拠を明示したうえで提案する
    • 不明点は着手前にまとめて質問する。曖昧なまま進めない
    • エラーが同じ方法で3回続いたら、別のアプローチを提案する
  • コード・ファイルの提供
    • コード修正時はファイル全体を提供する（間違い防止のため。2000行超は要相談）
    • GASコードを修正した場合は、コード全体をコピーボタン付きで必ず提示する
    • indexを出力するときは、管理資料も必ず同時に更新・出力する（漏れ防止）
    • コードには必ずコメントをつける
  • 説明・出力スタイル
    • 説明は結論 → 理由の順で書く
    • 忖度せず、リスクや懸念点を含めて率直に伝える
    • 制約や手間・注意点は先にセットで伝える（後出しにしない）
    • 選択肢を提示するときは、テキスト箇条書きではなくボタン式（ask_user_input）で表示する
  • スマホ確認方法
    • 実機確認推奨：iPhone SE（基準）→ iPhone 14（メイン）→ iPad Air（タブレット確認）
    • PCでの確認：DevToolsでスマホ表示にしてからGoFullPageでキャプチャ → Claudeに共有
    • DevToolsエミュレーションと実機は微妙にズレるため最終確認は実機で行う
  • 資料内メールアドレスの書き方
    • @ を &#64; と書くこと（Cloudflareの自動難読化を回避するため）

/* ================================================================
   定数
================================================================ */
const SHEET_NAME = "お知らせ";
const HIDE_DAYS  = 90;
const AUTO_FLAG  = "__auto__";

/* ================================================================
   空白行判定ヘルパー
================================================================ */
function isEmptyRow(row) {
  // E列=インデックス4（タイトル）、F列=インデックス5（本文）は列移動後も変わらず
  const e = String(row[4] || "").trim();
  const f = String(row[5] || "").trim();
  return e === "" && f === "";
}

/* ================================================================
   次の木曜日を返すヘルパー
================================================================ */
function getNextThursday(baseDate) {
  const d = new Date(baseDate);
  const day = d.getDay();
  const diff = day <= 4 ? 4 - day : 4 - day + 7;
  const add = diff === 0 ? 7 : diff;
  d.setDate(d.getDate() + add);
  return d;
}

/* ================================================================
   木曜日の日付を「○月○日（木）」形式タイトルに変換
================================================================ */
function formatThursdayTitle(date) {
  const m = date.getMonth() + 1;
  const d = date.getDate();
  return m + "月" + d + "日（木）は定休日のため、お休みさせていただきます。";
}

/* ================================================================
   __auto__ 行を管理する関数
   bannerOn: true=バナーオン / false=バナーオフ
================================================================ */
function upsertAutoRow(sheet, bannerOn) {
  const today = new Date();
  const nextThursday = getNextThursday(today);
  const title = formatThursdayTitle(nextThursday);
  const lastRow = sheet.getLastRow();

  let autoRowIndex = -1;
  if (lastRow >= 2) {
    // F列（列番号6）の__auto__フラグで自動行を検索
    const fCol = sheet.getRange(2, 6, lastRow - 1, 1).getValues();
    for (let i = 0; i < fCol.length; i++) {
      if (String(fCol[i][0]).trim() === AUTO_FLAG) {
        autoRowIndex = i + 2;
        break;
      }
    }
  }

  if (autoRowIndex === -1) {
    const insertRow = lastRow + 1;
    if (insertRow > sheet.getMaxRows()) {
      sheet.insertRowsAfter(sheet.getMaxRows(), 1);
    }
    autoRowIndex = insertRow;
  }

  // 新列構成：A(1)=投稿日, B(2)=バナー, C(3)=Web表示, D(4)=分類, E(5)=タイトル, F(6)=AUTO_FLAG, G(7)=URL
  sheet.getRange(autoRowIndex, 1).setValue(today);              // A列：投稿日
  sheet.getRange(autoRowIndex, 1).setNumberFormat("yyyy.mm.dd");
  sheet.getRange(autoRowIndex, 2).setValue(bannerOn);           // B列：バナー表示
  sheet.getRange(autoRowIndex, 3).setValue(true);               // C列：Web表示
  sheet.getRange(autoRowIndex, 4).setValue("定休日（自動）");   // D列：分類
  sheet.getRange(autoRowIndex, 5).setValue(title);              // E列：タイトル
  sheet.getRange(autoRowIndex, 6).setValue(AUTO_FLAG);          // F列：AUTO_FLAG
  sheet.getRange(autoRowIndex, 7).setValue("");                 // G列：URL
  // AUTO行は明るい緑（自動生成行の統一カラー）
  sheet.getRange(autoRowIndex, 1, 1, 7).setBackground("#d1fae5");
  sheet.getRange(autoRowIndex, 1, 1, 7).setFontColor("#065f46");
}

/* ================================================================
   テンプレート行（2〜6行目）を空白入力欄として確保
================================================================ */
function ensureInputRows(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;
  const allData  = sheet.getRange(2, 1, lastRow - 1, 7).getValues();
  const dataRows = allData.filter(row => !isEmptyRow(row));
  dataRows.sort((a, b) => {
    // A列(インデックス0)=投稿日でソート
    const da = a[0] instanceof Date ? a[0].getTime() : 0;
    const db = b[0] instanceof Date ? b[0].getTime() : 0;
    return db - da;
  });
  sheet.getRange(2, 1, lastRow - 1, 7).clearContent();
  sheet.getRange(2, 1, 5, 7).setBackground(null).setFontColor(null); // テンプレート行2〜6行目のみ色リセット
  // 新列構成：A(1)=投稿日, B(2)=バナー, C(3)=Web表示, D(4)=分類
  // DATE_NOTE定数（onEdit冒頭で定義）を参照して重複を排除
  for (let i = 2; i <= 6; i++) {
    sheet.getRange(i, 1).setNote(DATE_NOTE);     // A列：投稿日メモ
    sheet.getRange(i, 2).setValue(false);         // B列：バナー表示
    sheet.getRange(i, 3).setValue(false);         // C列：Web表示
    sheet.getRange(i, 4).setValue("----------"); // D列：分類
  }
  if (dataRows.length > 0) {
    const neededLastRow = 6 + dataRows.length;
    if (neededLastRow > sheet.getMaxRows()) {
      sheet.insertRowsAfter(sheet.getMaxRows(), neededLastRow - sheet.getMaxRows());
    }
    sheet.getRange(7, 1, dataRows.length, 7).setValues(dataRows);
  }
  // 全行（2行目以降）の垂直配置を上揃えに統一
  sheet.getRange(2, 1, sheet.getLastRow() - 1, 7).setVerticalAlignment("top");
}

/* ================================================================
   90日経過行をグレーアウト＋B列・C列チェックを自動オフ
   ※AUTO行（__auto__）は対象外
================================================================ */
function applyGrayOut(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 7) return;
  const today = new Date();
  for (let i = 7; i <= lastRow; i++) {
    const dateVal  = sheet.getRange(i, 1).getValue();                // A列：投稿日
    const autoFlag = String(sheet.getRange(i, 6).getValue()).trim(); // F列：AUTO_FLAG
    const range    = sheet.getRange(i, 1, 1, 7);
    // AUTO行はスキップ（緑色を維持）
    if (autoFlag === AUTO_FLAG) continue;
    if (dateVal instanceof Date) {
      const diff = (today - dateVal) / (1000 * 60 * 60 * 24);
      if (diff > HIDE_DAYS) {
        range.setBackground("#374151");
        range.setFontColor("#f9fafb");
        sheet.getRange(i, 2).setValue(false); // B列：バナー表示オフ
        sheet.getRange(i, 3).setValue(false); // C列：Web表示オフ
      } else {
        range.setBackground(null);
        range.setFontColor(null);
      }
    } else {
      // 日付なし行：既存の黒背景をリセット
      range.setBackground(null);
      range.setFontColor(null);
    }
  }
}

/* ================================================================
   doGet：WebサイトにJSONを返す（__auto__フラグは本文に表示しない）
   ⚠️ この関数を変更したときはデプロイが必要
================================================================ */
function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const BADGE_MAP = {
    "新商品入荷":     "new",
    "新商品":         "new",
    "誕生石":         "gem",
    "お知らせ":       "info",
    "臨時休業":       "closed",
    "定休日":         "holiday",
    "定休日（自動）": "holiday"
  };
  const today = new Date();
  const news = rows.slice(1)
    .filter(row => {
      if (row[2] !== true) return false;           // C列(2)：Web表示チェック
      if (!row[4] && !row[5]) return false;        // E列(4)：タイトル、F列(5)：本文
      if (row[0] instanceof Date) {                // A列(0)：投稿日
        const diff = (today - row[0]) / (1000 * 60 * 60 * 24);
        if (diff > HIDE_DAYS) return false;
      }
      return true;
    })
    .map(row => ({
      banner: row[1] === true ? "バナーに表示" : "",  // B列(1)：バナー表示
      badge:  row[3] && BADGE_MAP[row[3]] ? "badge-" + BADGE_MAP[row[3]] : "", // D列(3)：分類
      date:   row[0] instanceof Date               // A列(0)：投稿日
                ? Utilities.formatDate(row[0], "Asia/Tokyo", "yyyy.MM.dd")
                : row[0] ? String(row[0]).slice(0, 10).replace(/\//g, ".") : "",
      title:  String(row[4] || ""),                // E列(4)：タイトル
      text:   String(row[5] || "") === AUTO_FLAG ? "" : String(row[5] || ""), // F列(5)：本文
      url:    row[6] ? String(row[6]) : ""         // G列(6)：URL
    }))
    .sort((a, b) => b.date.localeCompare(a.date));
  return ContentService
    .createTextOutput(JSON.stringify(news))
    .setMimeType(ContentService.MimeType.JSON);
}

/* ================================================================
   onEdit：E列またはF列入力でA列に日付自動入力
            A列編集時にメモが消えていたら自動復元
================================================================ */
const DATE_NOTE = "投稿日は【自動入力】されます。\n変更したい場合はセルをクリックしてカレンダーから選んでください。";

function onEdit(e) {
  try {
    const sheet = e.source.getActiveSheet();
    if (sheet.getName() !== SHEET_NAME) return;
    const row = e.range.getRow();
    const col = e.range.getColumn();

    // E列(5)またはF列(6)入力時：A列(1)に日付自動入力
    if ((col === 5 || col === 6) && row >= 2) {
      const dateCell = sheet.getRange(row, 1); // A列
      if (!dateCell.getValue()) {
        dateCell.setValue(new Date());
        dateCell.setNumberFormat("yyyy.mm.dd");
      }
      // A列メモも復元
      if (!dateCell.getNote()) dateCell.setNote(DATE_NOTE);
    }

    // A列(1)編集時：メモが消えていたら自動復元
    if (col === 1 && row >= 2) {
      const dateCell = sheet.getRange(row, 1);
      if (!dateCell.getNote()) dateCell.setNote(DATE_NOTE);
    }
  } catch(err) {
    // エラーを無視して続行（GASログには記録）
    console.error("onEdit error:", err);
  }
}

/* ================================================================
   A列全行にメモを一括設定（初回のみ実行・以降はonEditで自動復元）
================================================================ */
function setNoteToAllDates() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;
  for (let i = 2; i <= lastRow; i++) {
    sheet.getRange(i, 1).setNote(DATE_NOTE); // A列
  }
  SpreadsheetApp.getUi().alert("完了\nA列 " + (lastRow - 1) + "行にメモを設定しました。");
}

/* ================================================================
   深夜一括処理（時間ベーストリガーで自動実行）
================================================================ */
function hideOldRows() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const today = new Date();
  const dow = today.getDay(); // 0=日,1=月,2=火,3=水,4=木
  if (dow === 2) upsertAutoRow(sheet, true);  // 火曜深夜：バナーオン
  if (dow === 4) upsertAutoRow(sheet, false); // 木曜深夜：バナーオフ・次週木曜に更新
  applyGrayOut(sheet);
  ensureInputRows(sheet);
  formatDateColumn(sheet); // A列日付フォーマット統一
}

/* ================================================================
   A列（投稿日）の日付フォーマットを yyyy.mm.dd に統一
   ※Dateオブジェクトの行のみ対象。文字列行はスキップ
================================================================ */
function formatDateColumn(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;
  const dateRange = sheet.getRange(2, 1, lastRow - 1, 1);
  dateRange.setNumberFormat("yyyy.mm.dd"); // 全行に一括適用
}
